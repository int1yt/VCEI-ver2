\begin{tikzpicture}[
  font=\small,
  >=Latex,

  blk/.style={

% Feedback loop
\draw[sig] (ids.south) -- ++(0,-0.7) -| node[pos=0.25,right,text=black!70]{告警元数据} (ring.east);
