\documentclass[tikz,border=10pt]{standalone}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\usetikzlibrary{arrows.meta, backgrounds}

\begin{document}
	\begin{tikzpicture}[font=\sffamily]
		\begin{axis}[
			width=12cm, height=6cm,
			xlabel={报文序列 (Message Sequence)},
			ylabel={累积偏移 $D_i$ ($\mu$s)},
			grid=both,
			grid style={line width=.1pt, draw=gray!10},
			legend pos=north west,
			axis line style={thick}
			]
			% 正常轨迹 (Normal S
			kew) - 稳定斜率
			\addplot[blue!70, thick, domain=0:50, samples=50] {0.05*x + rand*0.08};
			\addlegendentry{合法 ECU 指纹 (Normal)}
			
			% 攻击注入点
			\draw[red, dashed, thick] (axis cs:30,-1) -- (axis cs:30,5);
			\node[red, anchor=south, font=\footnotesize\bfseries] at (axis cs:30, 4.2) {攻击者注入 (Attack)};
			
			% 异常轨迹 (Abnormal Skew) - 斜率发生突变
			\addplot[red, ultra thick, domain=30:50, samples=20] {0.05*30 + 0.2*(x-30) + rand*0.1};
			\addlegendentry{注入攻击轨迹 (Anomalous)}
			
			% 斜率辅助线
			\draw[black, thick, ->] (axis cs:10,0.5) -- (axis cs:20,1.0) node[midway, below, font=\tiny] {斜率 $S_{ref}$};
		\end{axis}
	\end{tikzpicture}
